const poemas = [
  {
    id: "sorriso-silencio",
    titulo: "Quando o sorriso virou silêncio",
    texto: `Foi quando um sorriso em meu rosto tornou-se lágrimas
que percebi que a tristeza já não era passageira.

Ela ficou.
Fez morada.

E eu aprendi a conviver com o que antes eu chamava de fase.`
  },

  {
    id: "cicatrizes",
    titulo: "Cicatrizes",
    texto: `Cicatrizes não são apenas marcas na pele,
são histórias que o tempo não conseguiu apagar.

Algumas doem em silêncio,
outras gritam dentro de mim.`
  }
];

const container = document.getElementById("poemas-container");

poemas.forEach(p => {
  const div = document.createElement("div");
  div.classList.add("poema");

  div.innerHTML = `
    <h2>${p.titulo}</h2>
    <p>${p.texto}</p>
    <a href="poema.html?id=${p.id}" style="color: var(--accent);">
      ler mais →
    </a>
  `;

  container.appendChild(div);
});

/* TEMA */
const botaoTema = document.getElementById("toggleTema");

botaoTema.addEventListener("click", () => {
  document.body.classList.toggle("light");
});

/* MÚSICA */
const musica = document.getElementById("musica");
const btnMusica = document.getElementById("playMusica");

let tocando = false;

btnMusica.addEventListener("click", () => {
  if (!tocando) {
    musica.play();
    btnMusica.textContent = "⏸ parar música";
  } else {
    musica.pause();
    btnMusica.textContent = "🎧 música";
  }
  tocando = !tocando;
});
